```typescript
// packages/lattice-cli/src/io/sidecar-loader.ts

/** v1.1 sidecar wire format. The `version` field is REQUIRED and is the only accepted value. */
export interface SidecarFile {
  readonly version: "lattice-sidecar/v1";
  readonly task: string;
  /**
   * Outputs map. v1.1 supports ONLY the literal string "text" (text output)
   * and the built-in citation/artifact contract sentinels emitted by
   * `output.citations()` / `output.artifacts()` (serialized as
   * `{ "kind": "citations" }` / `{ "kind": "artifacts" }`).
   * Generic Standard Schema / Zod validators are NOT supported in v1.1
   * — the loader rejects them with kind="unsupported-output-shape".
   */
  readonly outputs: Record<string, SidecarOutputSpec>;
  readonly policy: import("lattice").PolicySpec;
  readonly contract: import("lattice").CapabilityContract;
}

/** Serializable output specs supported by v1.1. */
export type SidecarOutputSpec =
  | "text"
  | { readonly kind: "citations" }
  | { readonly kind: "artifacts" };

/** Discriminated load failure. Callers pattern-match on `kind`. */
export type SidecarLoadError =
  | { readonly kind: "file-not-found"; readonly path: string; readonly message: string }
  | { readonly kind: "malformed"; readonly path: string; readonly message: string }
  | { readonly kind: "version-mismatch"; readonly path: string; readonly message: string; readonly received: string }
  | { readonly kind: "unsupported-output-shape"; readonly path: string; readonly message: string; readonly outputKey: string };

export function isSidecarLoadError(value: unknown): value is SidecarLoadError;

/** Strict loader. Throws SidecarLoadError on any failure. */
export async function loadSidecar(path: string): Promise<SidecarFile>;

/** Pure projection — narrows a loaded sidecar to the fields materializeReplayEnvelope accepts.
 *  Returns the four optional fields ready to spread into MaterializeReplayEnvelopeOptions. */
export interface SidecarApplyResult {
  readonly task: string;
  readonly outputs: import("lattice").OutputContractMap;
  readonly policy: import("lattice").PolicySpec;
  readonly contract: import("lattice").CapabilityContract;
}
export function applySidecar(sidecar: SidecarFile): SidecarApplyResult;
```

```typescript
// packages/lattice-cli/src/io/sidecar-walker.ts

/** Pair of receipt + (optional) sidecar. `sidecar` is null when no sidecar file exists for this id. */
export interface WalkedReceiptWithSidecar {
  readonly id: string;
  readonly envelope: import("lattice").ReceiptEnvelope;
  readonly resolvedPath: string;
  readonly sidecar: SidecarFile | null;
  readonly sidecarPath: string | null;
}

/** Receipt-side error from the underlying walker — re-exported so the eval runner can switch on a single union. */
export type WalkedReceiptWithSidecarEntry =
  | WalkedReceiptWithSidecar
  | import("./receipt-walker.js").WalkedReceiptError;

export function isWalkedReceiptError(entry: WalkedReceiptWithSidecarEntry): entry is import("./receipt-walker.js").WalkedReceiptError;

/**
 * Walk `receiptsDir` (via walkReceiptsDirectory) and pair each yielded receipt
 * with its sidecar at `<sidecarDir>/<id>.json`. Missing-sidecar is NOT an
 * error here — the entry yields with `sidecar: null` and the consumer decides
 * how to react. Malformed-sidecar IS an error — the entry yields with the
 * receipt-walker's error shape `{ id, resolvedPath, error: SidecarLoadError }`
 * (the union is intentional: load failures from either side surface uniformly).
 *
 * Receipts-dir-missing propagates as the receipt walker's existing throw
 * (so the runner's existing exit-2 path still fires). Sidecars-dir-missing
 * is NON-FATAL: every entry yields `sidecar: null`.
 */
export async function* walkReceiptsWithSidecars(
  receiptsDir: string,
  sidecarDir: string,
): AsyncIterable<WalkedReceiptWithSidecarEntry>;
```
