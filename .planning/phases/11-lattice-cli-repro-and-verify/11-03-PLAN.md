Create `packages/lattice-cli/src/io/artifact-loader.ts`:

```ts
import { readFile, stat } from "node:fs/promises";
import { join, resolve } from "node:path";

import type { ArtifactInput } from "lattice";

const HASH_REGEX = /^[a-f0-9]{64}$/u;

export interface ArtifactLoaderError {
  readonly kind: "missing" | "invalid-hash";
  readonly hash: string;
  readonly path?: string;
  readonly message: string;
}

export function isArtifactLoaderError(value: unknown): value is ArtifactLoaderError {
  if (typeof value !== "object" || value === null) return false;
  const v = value as Record<string, unknown>;
  return (
    (v.kind === "missing" || v.kind === "invalid-hash") &&
    typeof v.hash === "string" &&
    typeof v.message === "string"
  );
}

export function createFilesystemArtifactLoader(
  fixturesDir: string,
): (hash: string) => Promise<ArtifactInput> {
  const root = resolve(fixturesDir);
  return async (hash: string): Promise<ArtifactInput> => {
    if (!HASH_REGEX.test(hash)) {
      throw {
        kind: "invalid-hash",
        hash,
        message: `Artifact hash ${hash!} is not a 64-char lowercase hex SHA-256 digest. Refusing to read.`,
      } satisfies ArtifactLoaderError;
    }
    const filePath = join(root, `${hash}.bin`);
    try {
      const [bytes, stats] = await Promise.all([
        readFile(filePath),
        stat(filePath),
      ]);
      const input: ArtifactInput = {
        id: hash,
        kind: "file",
        mime: "application/octet-stream",
        value: new Uint8Array(bytes),
        size: stats.size,
      };
      return input;
    } catch (err) {
      const message = err instanceof Error ? err.message : String(err);
      throw {
        kind: "missing",
        hash,
        path: filePath,
        message,
      } satisfies ArtifactLoaderError;
    }
  };
}
```

Notes:
- `ArtifactInput` fields: confirm shape from `packages/lattice/src/artifacts/artifact.ts` (the public type is re-exported from `"lattice"`). The example above uses `{ id, kind, mime, value, size }` — adjust to whatever the public `ArtifactInput` actually requires (the codebase uses `id` and `kind` + a `value` field; verify against the source once before writing). If `id` is required and must equal the hash, the assignment `id: hash` is correct.

- Reading and stat-ing in parallel via `Promise.all` is microscopic; serial would also be fine. Keep whichever ends up clearer.

```ts
import { readFile } from "node:fs/promises";
import { join, resolve } from "node:path";

import type { ReceiptEnvelope } from "lattice";

const PAYLOAD_TYPE = "application/vnd.lattice.receipt+json";

export interface ReceiptLoadError {
  readonly kind: "missing" | "malformed";
  readonly resolvedPath: string;
  readonly message: string;
}

export function isReceiptLoadError(value: unknown): value is ReceiptLoadError {
  if (typeof value !== "object" || value === null) return false;
  const v = value as Record<string, unknown>;
  return (
    (v.kind === "missing" || v.kind === "malformed") &&
    typeof v.resolvedPath === "string" &&
    typeof v.message === "string"
  );
}

function isReceiptEnvelopeShape(value: unknown): value is ReceiptEnvelope {
  if (typeof value !== "object" || value === null) return false;
  const v = value as Record<string, unknown>;
  return (
    v.payloadType === PAYLOAD_TYPE &&
    typeof v.payload === "string" &&
    Array.isArray(v.signatures)
  );
}

export interface ReceiptResolverOptions {
  readonly receiptsDir?: string;   // default: `.lattice/receipts/` relative to cwd
}

function resolveReceiptPath(target: string, options: ReceiptResolverOptions): string {
  const treatAsPath = target.includes("/") || target.endsWith(".json");
  if (treatAsPath) return resolve(target);
  const dir = options.receiptsDir ?? join(".lattice", "receipts");
  return resolve(join(dir, `${target}.json`));
}

export interface LoadedReceipt {
  readonly envelope: ReceiptEnvelope;
  readonly resolvedPath: string;
  readonly idOrPath: string;
}

export async function loadReceiptByIdOrPath(
  target: string,
  options: ReceiptResolverOptions = {},
): Promise<LoadedReceipt> {
  const resolvedPath = resolveReceiptPath(target, options);
  let text: string;
  try {
    text = await readFile(resolvedPath, "utf8");
  } catch (err) {
    const message = err instanceof Error ? err.message : String(err);
    throw { kind: "missing", resolvedPath, message } satisfies ReceiptLoadError;
  }
  let parsed: unknown;
  try {
    parsed = JSON.parse(text);
  } catch (err) {
    const message = err instanceof Error ? err.message : String(err);
    throw { kind: "malformed", resolvedPath, message } satisfies ReceiptLoadError;
  }
  if (!isReceiptEnvelopeShape(parsed)) {
    throw {
      kind: "malformed",
      resolvedPath,
      message: "Receipt JSON does not match ReceiptEnvelope shape (payloadType/payload/signatures).",
    } satisfies ReceiptLoadError;
  }
  return { envelope: parsed, resolvedPath, idOrPath: target };
}
```

```ts
import { defineCommand } from "citty";

import {
  decodeReceiptPayload,        // see note below if NOT public
  materializeReplayEnvelope,
  replayOffline,
  type CapabilityReceiptBody,
  type ReceiptEnvelope,
  type RunResult,
  verifyReceipt,
} from "lattice";

import { createFilesystemArtifactLoader, isArtifactLoaderError } from "../io/artifact-loader.js";
import { isKeysetLoadError, loadKeySetFromPath } from "../io/keyset-loader.js";
import { isReceiptLoadError, loadReceiptByIdOrPath } from "../io/receipt-loader.js";

interface ReproDeps {
  readonly stdout: (line: string) => void;
  readonly stderr: (line: string) => void;
  readonly exit: (code: number) => void;
}
const defaultDeps: ReproDeps = {
  stdout: (line) => process.stdout.write(line + "\n"),
  stderr: (line) => process.stderr.write(line + "\n"),
  exit: (code) => process.exit(code),
};

async function sha256Hex(text: string): Promise<string> {
  const bytes = new TextEncoder().encode(text);
  const digest = await crypto.subtle.digest("SHA-256", bytes);
  return Array.from(new Uint8Array(digest))
    .map((b) => b.toString(16).padStart(2, "0"))
    .join("");
}

function printSummary(
  body: CapabilityReceiptBody,
  verdict: "match" | "drift",
  deps: ReproDeps,
  diff?: { expected: string; actual: string },
): void {
  // Lean, redaction-safe summary. Order is stable so scripts can grep.
  deps.stdout(`receiptId=${body.receiptId}`);
  deps.stdout(`kid=${body.kid}`);
  deps.stdout(`contractVerdict=${body.contractVerdict}`);
  deps.stdout(`model.requested=${body.model.requested}`);
  deps.stdout(`route.providerId=${body.route.providerId}`);
  deps.stdout(`route.capabilityId=${body.route.capabilityId}`);
  deps.stdout(`usage.costUsd=${body.usage.costUsd ?? "null"}`);
  deps.stdout(`verdict=${verdict}`);
  if (verdict === "drift" && diff) {
    // First 200 chars of expected vs actual (best-effort diff).
    deps.stdout(`expected.outputHash=${diff.expected.slice(0, 200)}`);
    deps.stdout(`actual.outputHash=${diff.actual.slice(0, 200)}`);
  }
}

export async function runRepro(
  args: { target: string; key?: string; fixtures?: string; receiptsDir?: string },
  deps: ReproDeps = defaultDeps,
): Promise<void> {
  // Stage 1: load receipt.
  let envelope: ReceiptEnvelope;
  try {
    const loaded = await loadReceiptByIdOrPath(
      args.target,
      args.receiptsDir !== undefined ? { receiptsDir: args.receiptsDir } : {},
    );
    envelope = loaded.envelope;
  } catch (err) {
    const message = isReceiptLoadError(err) ? `${err.kind} at ${err.resolvedPath}: ${err.message}` : String(err);
    deps.stderr(`FAIL kind=receipt-load-failed reason=${message}`);
    deps.exit(2);
    return;
  }

  // Stage 2: load keyset.
  let keySet;
  try {
    keySet = await loadKeySetFromPath(args.key);
  } catch (err) {
    const message = isKeysetLoadError(err) ? `${err.kind} at ${err.path}: ${err.message}` : String(err);
    deps.stderr(`FAIL kind=keyset-load-failed reason=${message}`);
    deps.exit(2);
    return;
  }

  // Stage 3: build the filesystem loader. Default fixtures dir is cwd-relative
  // `.lattice/fixtures/` per CONTEXT.md.
  const fixturesDir = args.fixtures ?? ".lattice/fixtures";
  const artifactLoader = createFilesystemArtifactLoader(fixturesDir);

  // Stage 4: materialize. Phase 10's materializer verifies FIRST, so a
  // tampered receipt never invokes the artifact loader.
  let envelopeReplay;
  try {
    envelopeReplay = await materializeReplayEnvelope(envelope, {
      artifactLoader,
      keySet,
    });
  } catch (err) {
    // err is a MaterializationError (verify-failed | artifact-load-failed | envelope-malformed)
    // OR an ArtifactLoaderError surfaced through the loader callback.
    const kind =
      (typeof err === "object" && err !== null && typeof (err as { kind?: unknown }).kind === "string")
        ? (err as { kind: string }).kind
        : "unknown";
    const message =
      (typeof err === "object" && err !== null && typeof (err as { message?: unknown }).message === "string")
        ? (err as { message: string }).message
        : String(err);
    deps.stderr(`FAIL kind=${kind} reason=${message}`);
    deps.exit(2);
    return;
  }

  // We already know verifyReceipt passed inside materialize. Decode the body
  // again for the summary. Use a fresh verifyReceipt call to obtain the typed
  // body (avoids reaching into receipts/canonical.ts internals).
  const verifyResult = await verifyReceipt(envelope, keySet);
  if (!verifyResult.ok) {
    // Should be unreachable — materialize already ran verify-FIRST.
    deps.stderr(`FAIL kind=verify-failed reason=${verifyResult.error.message}`);
    deps.exit(2);
    return;
  }
  const body = verifyResult.body;

  // Stage 5: replay.
  const result: RunResult = await replayOffline(envelopeReplay);

  if (!result.ok) {
    deps.stderr(`FAIL kind=replay-failed reason=${result.error.kind}: ${result.error.message ?? ""}`);
    deps.exit(2);
    return;
  }

  // Stage 6: compare outputHash.
  if (body.outputHash === null) {
    deps.stderr(`FAIL kind=receipt-had-no-outputhash reason=Receipt has contractVerdict=${body.contractVerdict} and no recorded outputHash to diff against.`);
    deps.exit(2);
    return;
  }

  // Recompute the hash the same way Phase 9-04 commits to it:
  //   fingerprintArtifactValue(JSON.stringify(validation.outputs))
  // Which is sha256(utf8(JSON.stringify(outputs))) in hex.
  // `result.outputs` is the validated outputs map per replayOffline's success branch.
  const canonical = JSON.stringify(result.outputs);
  const actualHash = await sha256Hex(canonical);

  if (actualHash === body.outputHash) {
    printSummary(body, "match", deps);
    deps.exit(0);
    return;
  }
  printSummary(body, "drift", deps, { expected: body.outputHash, actual: actualHash });
  deps.exit(1);
}

export default defineCommand({
  meta: {
    name: "repro",
    description: "Verify a receipt, materialize a replay envelope from disk, replay it offline, and diff against the recorded outputHash.",
  },
  args: {
    target: {
      type: "positional",
      required: true,
      description: "Receipt id (resolved against .lattice/receipts/<id>.json) or a path containing '/' or ending in '.json'.",
    },

    },
```
