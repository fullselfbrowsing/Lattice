Purpose: `lattice verify` is the offline integrity check — a developer with only a receipt JSON file and a keyset file can prove (a) the signature is valid, (b) the canonical body hasn't drifted, (c) the signing `kid` is `active` in the keyset. This plan must do this without running the runtime, without touching artifact storage, and without any network. The verbose layered diff is reserved for `lattice repro` (plan 11-03); `verify` is the lean offline check.

Output: A working `verify` subcommand with mock-argv-driven integration tests, a tiny keyset file loader, and crisp single-line stdout/stderr output that scripts can parse.

- Output format: single line `OK kid=<kid> verdict=<verdict>` / `FAIL kind=... reason=...`.
- Default keyset path: `~/.lattice/keyset.json` resolved via `os.homedir()`.
- Exit codes: 0 = OK, 1 = verify FAIL, 2 = load/io failure (CONTEXT.md exit-code contract).
- Subcommand handler tested via mock argv (no spawn); the bin smoke test in plan 11-01 covers the spawn path.
- Redacted by default; no `--unsafe-unredacted` flag (CLI-05).

```ts
import { readFile } from "node:fs/promises";
import { homedir } from "node:os";
import { resolve, join } from "node:path";

import { createMemoryKeySet, type KeyEntry, type KeySet } from "lattice";

export interface KeysetLoadError {
  readonly kind: "missing" | "malformed";
  readonly path: string;
  readonly message: string;
}

const DEFAULT_RELATIVE = join(".lattice", "keyset.json");

export function defaultKeysetPath(): string {
  return join(homedir(), DEFAULT_RELATIVE);
}

export function expandTilde(p: string): string {
  if (p === "~") return homedir();
  if (p.startsWith("~/")) return join(homedir(), p.slice(2));
  return p;
}

function isKeyEntryShape(value: unknown): value is KeyEntry {
  if (typeof value !== "object" || value === null) return false;
  const v = value as Record<string, unknown>;
  if (typeof v.kid !== "string") return false;
  if (v.state !== "active" && v.state !== "retired" && v.state !== "revoked") return false;
  if (typeof v.publicKeyJwk !== "object" || v.publicKeyJwk === null) return false;
  return true;
}

export async function loadKeySetFromPath(rawPath?: string): Promise<KeySet> {
  const resolvedPath = resolve(expandTilde(rawPath ?? defaultKeysetPath()));
  let text: string;
  try {
    text = await readFile(resolvedPath, "utf8");
  } catch (err) {
    const message = err instanceof Error ? err.message : String(err);
    throw { kind: "missing", path: resolvedPath, message } satisfies KeysetLoadError;
  }
  let parsed: unknown;
  try {
    parsed = JSON.parse(text);
  } catch (err) {
    const message = err instanceof Error ? err.message : String(err);
    throw { kind: "malformed", path: resolvedPath, message } satisfies KeysetLoadError;
  }
  if (!Array.isArray(parsed) || !parsed.every(isKeyEntryShape)) {
    throw {
      kind: "malformed",
      path: resolvedPath,
      message: "Keyset file must be a JSON array of KeyEntry { kid, state, publicKeyJwk }.",
    } satisfies KeysetLoadError;
  }
  return createMemoryKeySet(parsed);
}

export function isKeysetLoadError(value: unknown): value is KeysetLoadError {
  return (
    typeof value === "object" &&
    value !== null &&
    (value as { kind?: unknown }).kind !== undefined &&
    ((value as { kind: unknown }).kind === "missing" ||
      (value as { kind: unknown }).kind === "malformed") &&
    typeof (value as { path?: unknown }).path === "string" &&
    typeof (value as { message?: unknown }).message === "string"
  );
}
```

```ts
import { readFile } from "node:fs/promises";
import { resolve } from "node:path";

import { defineCommand } from "citty";
import { verifyReceipt, type ReceiptEnvelope } from "lattice";

import {
  isKeysetLoadError,
  loadKeySetFromPath,
} from "../io/keyset-loader.js";

const PAYLOAD_TYPE = "application/vnd.lattice.receipt+json";

interface VerifyDeps {
  // Indirection so tests can inject a writeable stdout/stderr/exit.
  readonly stdout: (line: string) => void;
  readonly stderr: (line: string) => void;
  readonly exit: (code: number) => void;
}

const defaultDeps: VerifyDeps = {
  stdout: (line) => process.stdout.write(line + "\n"),
  stderr: (line) => process.stderr.write(line + "\n"),
  exit: (code) => process.exit(code),
};

function isReceiptEnvelopeShape(value: unknown): value is ReceiptEnvelope {
  if (typeof value !== "object" || value === null) return false;
  const v = value as Record<string, unknown>;
  return (
    v.payloadType === PAYLOAD_TYPE &&
    typeof v.payload === "string" &&
    Array.isArray(v.signatures)
  );
}

export async function runVerify(
  args: { receipt: string; key?: string },
  deps: VerifyDeps = defaultDeps,
): Promise<void> {
  // Step 1: load the keyset.
  let keySet;
  try {
    keySet = await loadKeySetFromPath(args.key);
  } catch (err) {
    const message = isKeysetLoadError(err) ? `${err.kind} at ${err.path}: ${err.message}` : String(err);
    deps.stderr(`FAIL kind=keyset-load-failed reason=${message}`);
    deps.exit(2);
    return;
  }

  // Step 2: load the receipt JSON.
  let envelope: ReceiptEnvelope;
  try {
    const resolved = resolve(args.receipt);
    const text = await readFile(resolved, "utf8");
    const parsed: unknown = JSON.parse(text);
    if (!isReceiptEnvelopeShape(parsed)) {
      throw new Error("Receipt JSON does not match ReceiptEnvelope shape (payloadType/payload/signatures).");
    }
    envelope = parsed;
  } catch (err) {
    const message = err instanceof Error ? err.message : String(err);
    deps.stderr(`FAIL kind=receipt-load-failed reason=${message}`);
    deps.exit(2);
    return;
  }

  // Step 3: verify. Surface ONLY the redacted body's kid + verdict on success.
  const result = await verifyReceipt(envelope, keySet);
  if (result.ok) {
    deps.stdout(`OK kid=${result.body.kid} verdict=${result.body.contractVerdict}`);
    deps.exit(0);
    return;
  }
  deps.stderr(`FAIL kind=${result.error.kind} reason=${result.error.message}`);
  deps.exit(1);
}

export default defineCommand({
  meta: {
    name: "verify",
    description: "Verify a receipt's signature and structural integrity against a keyset.",
  },
  args: {
    receipt: {
      type: "positional",
      required: true,
      description: "Path to a receipt JSON file (ReceiptEnvelope shape).",
    },
    key: {
      type: "string",
      description: "Path to the keyset JSON file (default: ~/.lattice/keyset.json).",
    },
  },
  async run({ args }) {
    await runVerify({ receipt: args.receipt, key: args.key });
  },
});
```

Notes:
- The handler is split into a named exported `runVerify(args, deps)` function plus the default-exported `defineCommand`. Tests import `runVerify` and inject a capturing `VerifyDeps`. This is the "subcommand handlers tested via mock argv (no spawn)" pattern from CONTEXT.md.
- `result.ok ? result.body.kid : result.error.kind` is the entire happy/sad surface — no extra fields leaked (CLI-05 redaction discipline).
- `verifyReceipt` is imported from `"lattice"` (public export). Do NOT reach into `lattice/src/*`.

- The `result.ok` branch must precede the FAIL branch — TypeScript narrows the union; do not destructure `result.error` outside the `!result.ok` block (verbatimModuleSyntax + exactOptionalPropertyTypes will flag it).
