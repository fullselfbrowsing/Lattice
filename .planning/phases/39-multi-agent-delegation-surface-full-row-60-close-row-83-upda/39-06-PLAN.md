export interface RunAgentCrewOptions {
  readonly root: AgentSpec;
  readonly hosts: { readonly childHost: AgentHost };
  readonly policy?: CrewPolicy;
  readonly signer?: ReceiptSigner;     // crew-level, threaded to every member (research OQ1)
  readonly tracer?: ...;               // match runAgent's tracer/pipeline option types
  readonly pipeline?: ...;
}
export interface CrewResult {
  readonly result: AgentResult;                          // parent's result, output untouched
  readonly perAgent: ReadonlyArray<{ readonly id: string; readonly usage: Usage; readonly iterations: number; readonly receiptCids: readonly string[] }>;
  readonly usage: Usage;                                  // crew aggregate
  readonly totalIterations: number;                       // across all agents (eval snapshot source)
  readonly receipts: ReadonlyArray<ReceiptEnvelope>;      // all minted envelopes incl. crew root
  readonly crewRootCid?: string;
}
export function runAgentCrew(options: RunAgentCrewOptions, config: LatticeConfig): Promise<CrewResult>;
