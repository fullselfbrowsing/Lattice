       3b. Add the helper at the bottom of the file (after normalizeAdapterUsage):
       ```ts
       async function hashInputArtifacts(
         artifacts: readonly ArtifactInput[],
       ): Promise<readonly string[]> {
         const out: string[] = [];
         for (const artifact of artifacts) {
           // ArtifactInput's value field carries the canonical bytes/value.
           // Coerce undefined → "" so we always produce a string entry per input.
           const fp = await fingerprintArtifactValue((artifact as { readonly value?: unknown }).value);
           out.push(fp?.value ?? "");
         }
         return out;
       }

       async function sha256HexOfCanonicalContract(
         contract: unknown,
       ): Promise<string | null> {
         if (contract === undefined || contract === null) return null;
         const canonical = canonicalize(contract);
         if (canonical === undefined) return null;
         const bytes = new TextEncoder().encode(canonical);
         const ab = new Uint8Array(bytes.byteLength);
         ab.set(bytes);
         const digest = await crypto.subtle.digest("SHA-256", ab.buffer as ArrayBuffer);
         return Array.from(new Uint8Array(digest))
           .map((b) => b.toString(16).padStart(2, "0"))
           .join("");
       }

       interface MaybeIssueReceiptInput {
         readonly runId: string;
         readonly contract?: CapabilityContract;
         readonly artifacts: readonly ArtifactInput[];
         readonly contractVerdict: ContractVerdict;
         readonly model: ReceiptModel;
         readonly route: ReceiptRoute;
         readonly usage: Usage;
         readonly outputs?: unknown;
         readonly noRouteReasons?: readonly RouteRejectReason[];
         readonly tripwireEvidence?: TripwireEvidence;
       }

       async function maybeIssueReceipt(
         normalized: NormalizedLatticeConfig,
         input: MaybeIssueReceiptInput,
       ): Promise<ReceiptEnvelope | undefined> {
         if (normalized.signer === undefined) return undefined;
         try {
           const inputHashes = await hashInputArtifacts(input.artifacts);
           const outputHash =
             input.outputs === undefined
               ? null
               : ((await fingerprintArtifactValue(input.outputs))?.value ?? null);
           const contractHash = await sha256HexOfCanonicalContract(input.contract);
           return await createReceipt(
             {
               runId: input.runId,
               model: input.model,
               route: input.route,
               usage: input.usage,
               contractVerdict: input.contractVerdict,
               contractHash,
               inputHashes,
               outputHash,
               ...(input.noRouteReasons !== undefined
                 ? { noRouteReasons: input.noRouteReasons }
                 : {}),
               ...(input.tripwireEvidence !== undefined
                 ? { tripwireEvidence: input.tripwireEvidence }
                 : {}),
             },
             normalized.signer,
           );
         } catch {
           // Receipt emission is best-effort. A signer failure must NOT crash
           // ai.run — the result already encodes the verdict.
           return undefined;
         }
       }
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| caller LatticeConfig.signer → runtime | A misconfigured signer (e.g., key with wrong kid embedded in publicKeyJwk) would emit unverifiable receipts. createInMemorySigner accepts the kid+publicKeyJwk separately so the caller cannot mismatch them without intent. |
| signer.sign() throw → ai.run | If the configured signer throws, the run MUST NOT crash. maybeIssueReceipt swallows the error and returns undefined; the RunResult.receipt is simply absent. |
| dist/index.d.ts → external consumers | Public surface promises (verifyReceipt, signer factories) become a SemVer commitment; we deliberately keep createReceipt internal so v1.2 can change its signature freely. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-23 | T (Tampering) | runWithConfig terminal branches | mitigate | Every terminal branch calls maybeIssueReceipt; the awk-style line-order check in acceptance criteria ensures emission cannot be skipped on any branch. |
| T-09-24 | R (Repudiation) | maybeIssueReceipt swallow-errors | accept | A signer that always throws produces NO receipt — caller can detect this from `result.receipt === undefined`. We do not log signer failures in v1.1; observability of the runtime path is the existing `events[]` mechanism. |
| T-09-25 | I (Information Disclosure) | inputHashes via fingerprintArtifactValue | mitigate | Artifacts are hashed via SHA-256 over their canonical bytes (existing storage/fingerprint.ts). Raw artifact bytes never enter the receipt — only hex digests. |
| T-09-26 | T (Tampering) | sha256HexOfCanonicalContract | mitigate | Contracts are canonicalized via canonicalize@3.0.0 BEFORE hashing — same JCS used for receipt body, so contractHash is reproducible by external verifiers given the contract. |
| T-09-27 | E (Elevation) | public-surface exports | mitigate | createReceipt, redactReceiptBody, canonicalizeReceiptBody, buildPae, encodeEnvelope, decodeEnvelope, importEd25519PrivateKey, importEd25519PublicKey are NOT exported. External consumers can only verify; they cannot forge receipt builders against private API. |
| T-09-28 | D (DoS) | maybeIssueReceipt | mitigate | Performance test (T13 in Task 1) asserts 100 receipts in under 5 seconds. fingerprintArtifactValue (existing) is bounded; canonicalize is bounded; Ed25519 sign is ~30k sig/s. |
