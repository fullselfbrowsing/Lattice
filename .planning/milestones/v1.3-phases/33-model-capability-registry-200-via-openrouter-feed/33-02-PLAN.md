    5. Pure exported helper `stripOpenRouterVariant`:
       ```typescript
       /**
        * Strip the OpenRouter variant suffix (`:free` or `:thinking`) from an
        * OpenRouter-shaped id (`vendor/model:variant`). Other adapter id shapes
        * pass through verbatim — does not, for example, alter `anthropic:claude-opus-4`.
        *
        * Exported because Phase 34 (quirks) and Phase 36 (sanitizers) need the
        * same normalization. Phase 33 D-11.
        */
       export function stripOpenRouterVariant(id: string): string {
         if (!OPENROUTER_VARIANT_RE.test(id)) return id;
         const colonIdx = id.lastIndexOf(":");
         return colonIdx === -1 ? id : id.slice(0, colonIdx);
       }
       ```

    6. Lazy Map builder (per RESEARCH §Pattern 1):
       ```typescript
       let _lookupCache: Map<string, ModelCapabilityProfile> | undefined;

       function getLookupMap(): Map<string, ModelCapabilityProfile> {
         if (_lookupCache !== undefined) return _lookupCache;
         const map = new Map<string, ModelCapabilityProfile>();
         // STATIC first so generated entries with the same canonical key
         // would overwrite — by current design STATIC and GENERATED entries
         // do not share keys (static profiles use direct adapters; generated
         // entries use the openrouter adapter), but the iteration order
         // documents the precedence.
         for (const p of STATIC_PROFILES) {
           map.set(`${p.adapter}:${p.id}`, p);
         }
         for (const p of GENERATED_PROFILES) {
           map.set(`${p.adapter}:${p.id}`, p);
         }
         _lookupCache = map;
         return map;
       }

       /** Test-only — reset the lazy cache between vitest cases. NOT exported from index.ts. */
       export function _resetLookupCacheForTests(): void {
         _lookupCache = undefined;
       }
       ```

    8. Public fuzzy lookup (D-10):
       ```typescript
       /**
        * Strip the OpenRouter variant suffix (if any) and return ALL matching
        * profiles across every adapter, in deterministic order: direct adapters
        * first (anthropic, openai, gemini, xai, openai-compat, lm-studio),
        * OpenRouter last. Useful for pre-routing capability inspection.
        */
       export function findCapabilityProfile(id: string): ModelCapabilityProfile[] {
         const stripped = stripOpenRouterVariant(id);
         const map = getLookupMap();
         const matches: ModelCapabilityProfile[] = [];
         for (const adapter of ADAPTER_ORDER) {
           const hit = map.get(`${adapter}:${stripped}`);
           if (hit) matches.push(hit);
         }
         return matches;
       }
       ```
