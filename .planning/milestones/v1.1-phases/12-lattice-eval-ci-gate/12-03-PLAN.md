    ```typescript
    /**
     * `lattice eval [--fixtures <dir>] [--baseline <path>] [--key <keyset>]
     *               [--judge-cache <dir>] [--init-baseline] [--cost-tolerance <n>]
     *               [--quality-tolerance <n>] [--judge-n <n>]`
     *
     * CI gate: walks .lattice/receipts/, replays each receipt offline, gates
     * baseline-relative cost and quality regressions with layered determinism
     * classes. stdout = one line of EvalRunReport JSON. stderr = human lines.
     *
     * Exit codes per CONTEXT.md exit-code matrix.
     */
    import { defineCommand } from "citty";

    import { isBaselineLoadError, writeBaseline, type Baseline, type BaselineEntry } from "../eval/baseline.js";
    import { runEvalSession, type EvalRunnerDeps } from "../eval/runner.js";
    import type { EvalConfig, EvalRunReport } from "../eval/types.js";
    import { isKeysetLoadError } from "../io/keyset-loader.js";
    import { isReceiptLoadError } from "../io/receipt-loader.js";

    export interface EvalDeps {
      readonly stdout: (line: string) => void;
      readonly stderr: (line: string) => void;
      readonly exit: (code: number) => void;
      readonly runSession?: typeof runEvalSession;
      readonly writeBaseline?: typeof writeBaseline;
      readonly now?: () => string;
      readonly runnerDeps?: EvalRunnerDeps;
    }

    const defaultDeps: EvalDeps = {
      stdout: (line) => process.stdout.write(line + "\n"),
      stderr: (line) => process.stderr.write(line + "\n"),
      exit: (code) => { process.exit(code); },
    };

    export interface RunEvalArgs {
      readonly fixtures?: string;
      readonly baseline?: string;
      readonly key?: string;
      readonly judgeCache?: string;
      readonly artifacts?: string;
      readonly initBaseline?: boolean;
      readonly costTolerance?: number;
      readonly qualityTolerance?: number;
      readonly judgeN?: number;
      readonly judgePrompt?: string;
    }

    function buildConfig(args: RunEvalArgs): EvalConfig {
      const config: EvalConfig = {
        fixturesDir: args.fixtures ?? ".lattice/receipts",
        baselinePath: args.baseline ?? ".lattice/baseline.json",
        judgeCacheDir: args.judgeCache ?? ".lattice/judge-cache",
        artifactsDir: args.artifacts ?? ".lattice/fixtures",
        costTolerance: args.costTolerance ?? 0.10,
        qualityTolerance: args.qualityTolerance ?? 0.05,
        judgeN: args.judgeN ?? 3,
        initBaseline: args.initBaseline ?? false,
        judgePrompt: args.judgePrompt ?? "Rate the quality of this output from 0 to 1.",
        ...(args.key !== undefined ? { keyPath: args.key } : {}),
      };
      return config;
    }

    function reportFailure(deps: EvalDeps, kind: string, reason: string, code: 1 | 2): void {
      deps.stderr(`FAIL kind=${kind} reason=${reason}`);
      deps.exit(code);
    }

    function readErrorMessage(value: unknown): string {
      if (value instanceof Error) return value.message;
      if (typeof value === "object" && value !== null) {
        const v = value as { message?: unknown };
        if (typeof v.message === "string") return v.message;
      }
      return String(value);
    }

    export async function runEval(args: RunEvalArgs, deps: EvalDeps = defaultDeps): Promise<void> {
      const config = buildConfig(args);
      const runSession = deps.runSession ?? runEvalSession;
      const writeBaselineFn = deps.writeBaseline ?? writeBaseline;

      // Run the session. Catch typed load errors and surface them via FAIL ...
      let report: EvalRunReport;
      try {
        const runnerDeps: EvalRunnerDeps = {
          ...(deps.runnerDeps ?? {}),
          ...(deps.now !== undefined ? { now: deps.now } : {}),
        };
        report = await runSession(config, runnerDeps);
      } catch (err) {
        if (isBaselineLoadError(err)) return reportFailure(deps, `baseline-${err.kind}`, `${err.path}: ${err.message}`, 2);
        if (isKeysetLoadError(err)) return reportFailure(deps, `keyset-${err.kind}`, `${err.path}: ${err.message}`, 2);
        if (isReceiptLoadError(err)) return reportFailure(deps, `receipt-${err.kind}`, `${err.resolvedPath}: ${err.message}`, 2);
        return reportFailure(deps, "session-failed", readErrorMessage(err), 2);
      }

      // --init-baseline: write new baseline and exit 0.
      if (config.initBaseline) {
        const entries: Record<string, BaselineEntry> = {};
        for (const f of report.fixtures) {
          if (f.verdict === "load-failed" || f.usage === null) continue;
          entries[f.fixtureId] = {
            usage: f.usage,
            qualityFloor: f.qualityScore !== null ? { score: f.qualityScore } : null,
          };
        }
        const baseline: Baseline = {
          version: "lattice-eval/v1",
          recordedAt: (deps.now ?? (() => new Date().toISOString()))(),
          fixtures: entries,
        };
        try {
          await writeBaselineFn(config.baselinePath, baseline);
        } catch (err) {
          return reportFailure(deps, "baseline-write-failed", readErrorMessage(err), 2);
        }
        // Emit the report anyway so consumers see what was recorded; exit 0.
        const finalReport: EvalRunReport = { ...report, exitCode: 0 };
        emitReport(finalReport, deps);
        deps.exit(0);
        return;
      }

      // Determine exit code.
      const exitCode: 0 | 1 = report.summary.regressed > 0 ? 1 : 0;
      const finalReport: EvalRunReport = { ...report, exitCode };
      emitReport(finalReport, deps);
      deps.exit(exitCode);
    }

    function emitReport(report: EvalRunReport, deps: EvalDeps): void {
      // stderr: one line per fixture, plus aggregate summary
      for (const f of report.fixtures) {
        deps.stderr(
          `${f.fixtureId} verdict=${f.verdict} regressionKind=${f.regressionKind ?? "none"} deltaCostPct=${f.deltaCostPct ?? "null"} deltaQuality=${f.deltaQuality ?? "null"}`,
        );
      }
      deps.stderr(
        `SUMMARY total=${report.summary.total} passed=${report.summary.passed} regressed=${report.summary.regressed} newFixtures=${report.summary.newFixtures}`,
      );
      // stdout: ONE line of JSON
      deps.stdout(JSON.stringify(report));
    }

    export default defineCommand({
      meta: {
        name: "eval",
        description:
          "Walk a fixtures directory of receipts, replay each, and gate baseline-relative cost/quality regressions for CI.",
      },
      args: {
        fixtures: { type: "string", description: "Directory of receipts (default: .lattice/receipts/)." },
        baseline: { type: "string", description: "Baseline JSON path (default: .lattice/baseline.json)." },
        key: { type: "string", description: "Keyset JSON path (default: ~/.lattice/keyset.json)." },
        "judge-cache": { type: "string", description: "Judge cache directory (default: .lattice/judge-cache/)." },
        artifacts: { type: "string", description: "Artifact bodies directory (default: .lattice/fixtures/)." },
        "init-baseline": { type: "boolean", description: "Write a new baseline from this run instead of gating." },
        "cost-tolerance": { type: "string", description: "Cost regression tolerance (default 0.10)." },
        "quality-tolerance": { type: "string", description: "Quality regression tolerance (default 0.05)." },
        "judge-n": { type: "string", description: "Judge repetitions for N=median (default 3)." },
        "judge-prompt": { type: "string", description: "Judge prompt (default: a generic rating prompt)." },
      },
      async run({ args }) {
        const callArgs: RunEvalArgs = {
          ...(args.fixtures !== undefined ? { fixtures: args.fixtures } : {}),
          ...(args.baseline !== undefined ? { baseline: args.baseline } : {}),
          ...(args.key !== undefined ? { key: args.key } : {}),
          ...(args["judge-cache"] !== undefined ? { judgeCache: args["judge-cache"] } : {}),
          ...(args.artifacts !== undefined ? { artifacts: args.artifacts } : {}),
          ...(args["init-baseline"] === true ? { initBaseline: true } : {}),
          ...(args["cost-tolerance"] !== undefined ? { costTolerance: Number(args["cost-tolerance"]) } : {}),
          ...(args["quality-tolerance"] !== undefined ? { qualityTolerance: Number(args["quality-tolerance"]) } : {}),
          ...(args["judge-n"] !== undefined ? { judgeN: Number(args["judge-n"]) } : {}),
          ...(args["judge-prompt"] !== undefined ? { judgePrompt: args["judge-prompt"] } : {}),
        };
        await runEval(callArgs);
      },
    });
    ```
