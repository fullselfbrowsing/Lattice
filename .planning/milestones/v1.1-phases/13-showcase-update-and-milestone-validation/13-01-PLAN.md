// Runtime construction
export function createAI(config: LatticeConfig): AI;
export type LatticeConfig = {
  sessions?: SessionStore;
  providers: ProviderAdapter[];
  signer?: ReceiptSigner;        // RECEIPT-10: when present, receipts are emitted
  // ...existing v1.0 fields
};

// Contracts (CONTRACT-01..06)
export function contract(input: CapabilityContractInput): CapabilityContract;
// Usage: contract({ budget: { maxCostUsd: 0.01 }, invariants: [inv.noPII("answer")] })

// Invariants (TRIP-01)
export const inv: {
  noPII: (path: string, opts?: InvariantOptions) => NoPiiInvariant;
  mustCite: (...) => MustCiteInvariant;
  fieldFromTable: (...) => FieldFromTableInvariant;
  matches: (schema: StandardSchemaV1) => MatchesInvariant;
};

// Signer + keyset (RECEIPT-03, RECEIPT-05, RECEIPT-10)
export function generateEd25519KeyPairJwk(): Promise<{
  kid: string;
  publicKeyJwk: JsonWebKey;
  privateKeyJwk: JsonWebKey;
}>;
export function createInMemorySigner(input: {
  kid: string;
  privateKeyJwk: JsonWebKey;
  publicKeyJwk: JsonWebKey;
}): ReceiptSigner;
export function createMemoryKeySet(entries: KeyEntry[]): KeySet;

// Receipt envelope shape (RECEIPT-01, RECEIPT-03)
export type ReceiptEnvelope = {
  payloadType: "application/vnd.lattice.receipt+json";
  payload: string;            // base64url(canonical JSON of CapabilityReceiptBody)
  signatures: ReceiptSignature[];
};
export type CapabilityReceiptBody = {
  receiptId: string;
  runId: string;
  // ...
  contractVerdict: "pass" | "tripwire-violated" | "no-contract-match" | "execution-failed";
  usage: { promptTokens: number; completionTokens: number; costUsd: number | string };
  inputHashes: readonly string[];
  outputHash: string | null;
  kid: string;
  // ...
};

// Run results carry an optional receipt (Phase 9-04 wiring)
export type RunSuccess = { ok: true; outputs: ...; receipt?: ReceiptEnvelope; ... };
export type RunFailure = { ok: false; error: LatticeRunError; receipt?: ReceiptEnvelope; ... };
export type LatticeRunError =
  | { kind: "tripwire-violated"; invariantId: string; evidence: TripwireEvidence; ... }
  | { kind: "no-contract-match"; reasons: ContractRejectReasonCode[]; ... }
  | { kind: "execution-failed"; ... };

// Providers (v1.0 reuse)
export function createFakeProvider(opts: { id: string; response: { rawOutputs: any } }): ProviderAdapter;

3. `export async function writeArtifactContentAddressed(fixturesDir, bytes)` — writes `bytes` (Buffer or string) to `<fixturesDir>/<sha256hex>.bin` using Node `crypto.createHash("sha256")`, returns the sha256 hex. Used by scenarios to persist artifact bodies BEFORE `ai.run` so `lattice repro` can rehydrate. Make idempotent (if the file already exists, no-op).

4. `export async function writeReceipt(receiptsDir, envelope)` — decodes `envelope.payload` (base64url → JSON.parse) to read `body.receiptId`, then `writeFile(<receiptsDir>/<receiptId>.json, JSON.stringify(envelope, null, 2))`. Returns the absolute path written.
