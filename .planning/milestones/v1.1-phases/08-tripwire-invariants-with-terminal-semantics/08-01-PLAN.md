    import type { StandardSchemaV1 } from "@standard-schema/spec";
    import type {
      InvariantDeclaration,
      FieldFromTableInvariant,
      MatchesInvariant,
      MustCiteInvariant,
      NoPiiInvariant,
    } from "./invariants.js";
    import { defaultPiiDetectors, type PiiDetector } from "./pii-detectors.js";

    export interface TripwireEvidence {
      readonly invariantId: string;
      readonly kind: "must-cite" | "field-from-table" | "no-pii" | "matches";
      readonly path: string;
      readonly observed: unknown;
      readonly message: string;
    }

    export type TripwireResult =
      | { readonly ok: true }
      | { readonly ok: false; readonly evidence: TripwireEvidence };

    export async function evaluateTripwires(
      output: unknown,
      invariants: readonly InvariantDeclaration[],
      detectors: readonly PiiDetector[] = defaultPiiDetectors,
    ): Promise<TripwireResult> {
      for (const declaration of invariants) {
        const result = await evaluateOne(output, declaration, detectors);
        if (!result.ok) return result;
      }
      return { ok: true };
    }

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Provider response → tripwire evaluator | Untrusted LLM output crosses into invariant evaluation. May contain attacker-controlled strings designed to evade detectors. |
| Caller-supplied StandardSchema (inv.matches) | Caller-provided validator runs against output. Schema author trust = caller trust. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-01 | Information Disclosure | tripwire.ts no-pii evidence | mitigate | `observed` carries only `{ detector, substring }` — never the full input field. Test asserts substring length ≤ input length and substring is a substring of the input. Phase 9 receipts will sign this evidence; leaking the full PII into evidence would defeat redact-before-sign. |
| T-08-02 | Tampering | pii-detectors.ts Luhn | mitigate | Luhn validates check digit; attacker-supplied 16-digit strings without valid Luhn (e.g. random IDs) do NOT trigger credit-card detector — reduces false positives that would erode trust per Pitfall #5. Test fixture asserts `4111111111111112` (invalid Luhn) does NOT match. |
| T-08-03 | Denial of Service | tripwire.ts path resolver `[*]` | accept | `[*]` materializes the array; pathological inputs (deeply nested wildcard chains) could allocate O(N^k). Accept for v1.1 — provider responses are bounded by output token caps. Document in tripwire.ts comment. |
| T-08-04 | Repudiation | evaluateTripwires impurity | mitigate | Evaluator is a pure function (no I/O, no Date.now, no random). Same (output, invariants) → same result. Test runs twice and asserts deep-equal results. Phase 9 receipts can reconstruct verdict deterministically. |
| T-08-05 | Spoofing | inv.matches schema injection | accept | Caller supplies the StandardSchema; if the schema's `~standard.validate` returns spurious success, evaluator trusts it. This is by design — `matches` is a caller-driven escape hatch. Document in inv.matches JSDoc. |
</threat_model>
