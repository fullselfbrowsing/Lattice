    ```typescript
    import type {
      ProviderGatewayMetadata,

      ProviderStream,

    import type { UsageRecord } from "../plan/plan.js";
    import type { ArtifactInput, ArtifactRef } from "../artifacts/artifact.js";
    import type { ValidatedToolCall } from "../tools/tool-call-validation.js";

    export interface CollectStreamOptions {
      readonly defaultOutput?: string;
    }

    export async function collectStream(
      stream: ProviderStream,
      options: CollectStreamOptions = {},
    ): Promise<ProviderRunResponse> {
      // implementation described below
    }
    ```

    Implement `collectStream()` with these exact semantics:
    1. Track `chunkCount`.
    2. Track text deltas in `Map<string, string[]>`; when a `text-delta` chunk has no `output`, use `options.defaultOutput ?? "text"`.
    3. Track explicit output chunks in `rawOutputs[chunk.output] = chunk.value`.
    4. Track usage from `usage` chunks and `complete` chunks. Last explicit `usage` and last explicit `normalizedUsage` win.
    5. Track gateway metadata from `gateway` chunks and `complete.gateway`. Last gateway value wins.
    6. Track tool calls from `tool-call` chunks and `complete.toolCalls`.
    7. Track artifact refs from `complete.artifactRefs`.
    8. After iteration, fill text outputs into `rawOutputs` only for output names not already set by `output` or `complete.rawOutputs`.
    9. Merge `complete.rawOutputs` into `rawOutputs` when present, with complete values winning.
    10. Return a `ProviderRunResponse` with conditional spreads for `artifactRefs`, `usage`, `normalizedUsage`, `gateway`, `toolCalls`, and `rawResponse`.
    11. If no `complete.rawResponse` was supplied, set `rawResponse` to this content-free summary:
        ```typescript
        { kind: "lattice-stream-summary", chunkCount, outputNames: Object.keys(rawOutputs) }
        ```

    Create `packages/lattice/src/providers/streaming.test.ts` with tests named:
    - `collectStream assembles text deltas into the default output`
    - `collectStream lets explicit output chunks override accumulated text`
    - `collectStream merges usage gateway artifacts and tool calls from complete chunks`
    - `legacy ProviderAdapter literals still satisfy the optional stream contract`
