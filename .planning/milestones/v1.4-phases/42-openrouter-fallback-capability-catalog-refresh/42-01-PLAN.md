    Add a local helper in `openrouter.ts`:
    ```typescript
    function normalizeFallbackModels(models: readonly string[] | undefined): readonly string[] | undefined {
      if (models === undefined) return undefined;
      const normalized = models.map((model) => model.trim()).filter((model) => model.length > 0);
      return normalized.length > 0 ? normalized : undefined;
    }
    ```

    Add a local helper to extract observed model from `rawResponse`:
    ```typescript
    function observedModelFromRawResponse(rawResponse: unknown): string | undefined {
      if (typeof rawResponse !== "object" || rawResponse === null || Array.isArray(rawResponse)) {
        return undefined;
      }
      const model = (rawResponse as Record<string, unknown>).model;
      return typeof model === "string" ? model : undefined;
    }
    ```

    In `createOpenRouterProvider`, compute:
    ```typescript
    const fallbackModels = normalizeFallbackModels(options.fallbackModels);
    ```

    Do not add `fallbackModels` to `OpenAICompatibleProviderOptions`. Keep the generic OpenAI-compatible provider free of OpenRouter-specific public fields.

  <action>
    In `createOpenRouterProvider`, wrap the fetch passed to `createOpenAICompatibleProvider`.

    Add this local function after `fallbackModels` is computed:
    ```typescript
    const executeFetch: typeof fetch = fallbackModels === undefined
      ? fetchImpl
      : (async (url, init) => {
          if (typeof init?.body !== "string") {
            return fetchImpl(url, init);
          }
          const body = JSON.parse(init.body) as Record<string, unknown>;
          return fetchImpl(url, {
            ...init,
            body: JSON.stringify({
              ...body,
              models: [...fallbackModels],
            }),
          });
        }) as typeof fetch;
    ```

    Pass `fetch: executeFetch` to `createOpenAICompatibleProvider` instead of passing `options.fetch` through implicitly:
    ```typescript
    const baseAdapter = createOpenAICompatibleProvider({

      id: options.id ?? "openrouter",
      baseUrl,
      fetch: executeFetch,
    });
    ```
